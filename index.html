Basic installation


						composer require 'monolog/monolog'
					

						composer remove 'monolog/monolog'
composer.json 

{
    "require": {
        "monolog/monolog": "^2.4.*"
    }
}

Install dev packages

composer.json 

{
    "require": {
        "monolog/monolog": "^2.4"
    },
    "require-dev": {
        "phpunit/phpunit": "^9.5"
    }
}
